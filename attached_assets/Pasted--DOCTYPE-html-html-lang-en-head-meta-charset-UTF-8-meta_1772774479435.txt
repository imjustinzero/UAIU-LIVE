<script>
// ── State ─────────────────────────────────────────────────────────────
let ADMIN_KEY = '';
let rejectTarget = null;

// ── PIN Logic ─────────────────────────────────────────────────────────
(function() {
  // Support ?key= in URL
  const urlKey = new URLSearchParams(location.search).get('key');
  if (urlKey) { ADMIN_KEY = urlKey; unlock(); }

  document.getElementById('pin-input').addEventListener('keydown', e => {
    if (e.key === 'Enter') submitPin();
  });
})();

function submitPin() {
  const val = document.getElementById('pin-input').value.trim();
  if (!val) { document.getElementById('pin-error').textContent = 'Please enter your admin key.'; return; }
  ADMIN_KEY = val;
  // Validate by hitting health endpoint
  fetch(`/api/admin/health-check?admin_key=${encodeURIComponent(ADMIN_KEY)}`)
    .then(r => {
      if (r.status === 403) throw new Error('Invalid admin key.');
      return r.json();
    })
    .then(() => unlock())
    .catch(e => {
      document.getElementById('pin-error').textContent = e.message;
      ADMIN_KEY = '';
    });
}

function unlock() {
  document.getElementById('pin-gate').style.display = 'none';
  document.getElementById('app').style.display = 'flex';
  updateTime();
  setInterval(updateTime, 30000);
  loadListings();
}

function updateTime() {
  document.getElementById('header-time').textContent =
    'Last updated: ' + new Date().toLocaleTimeString();
}

// ── Tabs ──────────────────────────────────────────────────────────────
function switchTab(name) {
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));
  event.target.closest('.tab').classList.add('active');
  document.getElementById('panel-' + name).classList.add('active');
  if (name === 'listings') loadListings();
  if (name === 'webhooks') loadWebhooks();
  if (name === 'health')   loadHealth();
}

// ── Toast ─────────────────────────────────────────────────────────────
let toastTimer;
function toast(msg, type = 'ok') {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.className = 'show ' + type;
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => el.className = '', 3500);
}

// ── API helpers ───────────────────────────────────────────────────────
function api(path, opts = {}) {
  const sep = path.includes('?') ? '&' : '?';
  return fetch(`${path}${sep}admin_key=${encodeURIComponent(ADMIN_KEY)}`, {
    headers: { 'Content-Type': 'application/json' },
    ...opts,
  }).then(async r => {
    const data = await r.json().catch(() => ({}));
    if (!r.ok) throw new Error(data.error || `HTTP ${r.status}`);
    return data;
  });
}

// ── LISTINGS ──────────────────────────────────────────────────────────
async function loadListings() {
  const el = document.getElementById('listings-container');
  el.innerHTML = '<div class="empty"><span class="spinner"></span></div>';
  try {
    const listings = await api('/api/admin/listings/pending');
    document.getElementById('count-listings').textContent = listings.length;
    if (!listings.length) {
      el.innerHTML = '<div class="empty">✅ No pending listings — all clear.</div>';
      return;
    }
    el.innerHTML = listings.map(l => renderListing(l)).join('');
  } catch(e) {
    el.innerHTML = `<div class="empty" style="color:var(--red)">Error: ${e.message}</div>`;
  }
}

function renderListing(l) {
  const priceNum  = parseFloat(l.asking_price_per_tonne || l.askingPricePerTonne) || 0;
  const volNum    = parseFloat(l.volume_tonnes || l.volumeTonnes) || 0;
  const totalEur  = priceNum * volNum;
  const dateStr   = l.created_at || l.createdAt
    ? new Date(l.created_at || l.createdAt).toLocaleDateString() : '—';

  return `
  <div class="card" id="listing-${l.id}">
    <div class="card-header">
      <div>
        <div class="card-title">${esc(l.org_name || l.orgName || '—')} — ${esc(l.credit_type || l.creditType || '—')}</div>
        <div class="card-sub">Submitted ${dateStr} · ID: <span class="mono">${l.id}</span></div>
      </div>
      <div class="card-actions">
        <button class="btn btn-green btn-sm" onclick="approveL('${l.id}')">✓ Approve</button>
        <button class="btn btn-red btn-sm" onclick="openRejectModal('${l.id}')">✗ Reject</button>
      </div>
    </div>
    <div class="card-body">
      <div class="grid-3">
        <div class="kv"><span class="kv-label">Standard</span><span class="kv-value">${esc(l.standard || '—')}</span></div>
        <div class="kv"><span class="kv-label">Volume</span><span class="kv-value">${volNum.toLocaleString()} t CO₂</span></div>
        <div class="kv"><span class="kv-label">Ask Price</span><span class="kv-value">€${priceNum.toFixed(2)}/t</span></div>
        <div class="kv"><span class="kv-label">Total Value</span><span class="kv-value">€${totalEur.toLocaleString(undefined,{maximumFractionDigits:0})}</span></div>
        <div class="kv"><span class="kv-label">Project Origin</span><span class="kv-value">${esc(l.project_origin || l.projectOrigin || '—')}</span></div>
        <div class="kv"><span class="kv-label">Registry Serial</span><span class="kv-value mono">${esc(l.registry_serial || l.registrySerial || 'Not provided')}</span></div>
        <div class="kv"><span class="kv-label">Contact</span><span class="kv-value">${esc(l.contact_name || l.contactName || '—')}</span></div>
        <div class="kv"><span class="kv-label">Seller Email</span><span class="kv-value">${esc(l.email || '—')}</span></div>
        <div class="kv"><span class="kv-label">Status</span><span class="kv-value"><span class="badge badge-yellow">PENDING</span></span></div>
      </div>
      ${l.additional_notes || l.additionalNotes ? `<div style="margin-top:12px;padding:10px 14px;background:#0a1525;border-radius:6px;font-size:12px;color:var(--muted)">${esc(l.additional_notes || l.additionalNotes)}</div>` : ''}
    </div>
  </div>`;
}

async function approveL(id) {
  const btn = event.target;
  btn.disabled = true; btn.textContent = '…';
  try {
    const res = await api(`/api/admin/listings/${id}/approve`, { method: 'POST', body: JSON.stringify({}) });
    toast(`✅ Listing approved and live — ${res.listing?.name || id}`, 'ok');
    document.getElementById(`listing-${id}`)?.remove();
    const rem = document.querySelectorAll('[id^="listing-"]').length;
    document.getElementById('count-listings').textContent = rem;
    if (!rem) document.getElementById('listings-container').innerHTML =
      '<div class="empty">✅ No pending listings — all clear.</div>';
  } catch(e) {
    toast('Error: ' + e.message, 'err');
    btn.disabled = false; btn.textContent = '✓ Approve';
  }
}

function openRejectModal(id) {
  rejectTarget = id;
  document.getElementById('reject-modal').classList.add('open');
  document.getElementById('reject-reason').focus();
}
function closeModal() {
  rejectTarget = null;
  document.getElementById('reject-modal').classList.remove('open');
}
async function confirmReject() {
  if (!rejectTarget) return;
  const reason = document.getElementById('reject-reason').value.trim();
  const btn = document.getElementById('confirm-reject-btn');
  btn.disabled = true; btn.textContent = 'Sending…';
  try {
    await api(`/api/admin/listings/${rejectTarget}/reject`, {
      method: 'POST',
      body: JSON.stringify({ reason }),
    });
    toast('Listing rejected — seller notified via email.', 'ok');
    document.getElementById(`listing-${rejectTarget}`)?.remove();
    const rem = document.querySelectorAll('[id^="listing-"]').length;
    document.getElementById('count-listings').textContent = rem;
    if (!rem) document.getElementById('listings-container').innerHTML =
      '<div class="empty">✅ No pending listings — all clear.</div>';
    closeModal();
  } catch(e) {
    toast('Error: ' + e.message, 'err');
  } finally {
    btn.disabled = false; btn.textContent = 'Send Rejection';
  }
}

// ── WEBHOOKS ──────────────────────────────────────────────────────────
async function loadWebhooks() {
  const el = document.getElementById('webhooks-container');
  el.innerHTML = '<div class="empty"><span class="spinner"></span></div>';
  try {
    const failures = await api('/api/admin/webhooks/failures');
    document.getElementById('count-webhooks').textContent = failures.length;
    if (!failures.length) {
      el.innerHTML = '<div class="empty">✅ No unresolved webhook failures.</div>';
      return;
    }
    el.innerHTML = failures.map(f => renderWebhookFailure(f)).join('');
  } catch(e) {
    el.innerHTML = `<div class="empty" style="color:var(--red)">Error: ${e.message}</div>`;
  }
}

function renderWebhookFailure(f) {
  const dateStr = f.created_at || f.createdAt
    ? new Date(f.created_at || f.createdAt).toLocaleString() : '—';
  const retries = f.retry_count || f.retryCount || 0;
  const evType  = f.event_type || f.eventType || 'unknown';
  const tradeId = f.trade_id || f.tradeId || '—';
  const piId    = f.payment_intent_id || f.paymentIntentId || null;

  return `
  <div class="card" id="wh-${f.id}">
    <div class="card-header">
      <div>
        <div class="card-title"><span class="mono">${esc(evType)}</span></div>
        <div class="card-sub">Failed ${dateStr} · ${retries} retry attempt${retries===1?'':'s'}</div>
      </div>
      <div class="card-actions">
        <button class="btn btn-blue btn-sm" onclick="retryWebhook('${f.id}')">↻ Retry</button>
      </div>
    </div>
    <div class="card-body">
      <div class="grid-2" style="margin-bottom:12px">
        <div class="kv"><span class="kv-label">Trade ID</span><span class="kv-value mono">${esc(tradeId)}</span></div>
        <div class="kv"><span class="kv-label">Payment Intent ID</span><span class="kv-value mono">${piId ? esc(piId) : '—'}</span></div>
        <div class="kv"><span class="kv-label">Event ID</span><span class="kv-value mono">${esc(f.event_id || f.eventId || '—')}</span></div>
        <div class="kv"><span class="kv-label">Retry Count</span><span class="kv-value">${retries}</span></div>
      </div>
      <div style="background:#0a1525;border-radius:6px;padding:10px 14px">
        <div class="kv-label" style="margin-bottom:4px">Error Message</div>
        <div style="font-size:12px;color:var(--red);font-family:'Courier New',monospace;word-break:break-all">${esc(f.error_message || f.errorMessage || 'No message')}</div>
      </div>
    </div>
  </div>`;
}

async function retryWebhook(id) {
  const btn = event.target;
  btn.disabled = true; btn.textContent = '…';
  try {
    const res = await api(`/api/admin/webhooks/retry/${id}`, { method: 'POST', body: JSON.stringify({}) });
    if (res.action === 'captured') {
      toast(`✅ Captured! Gross: €${res.gross_eur?.toLocaleString()}`, 'ok');
    } else {
      toast('Resolved (no capture needed).', 'ok');
    }
    document.getElementById(`wh-${id}`)?.remove();
    const rem = document.querySelectorAll('[id^="wh-"]').length;
    document.getElementById('count-webhooks').textContent = rem;
    if (!rem) document.getElementById('webhooks-container').innerHTML =
      '<div class="empty">✅ No unresolved webhook failures.</div>';
  } catch(e) {
    toast('Retry failed: ' + e.message, 'err');
    btn.disabled = false; btn.textContent = '↻ Retry';
  }
}

// ── HEALTH ────────────────────────────────────────────────────────────
async function loadHealth() {
  const el = document.getElementById('health-container');
  el.innerHTML = '<div class="empty"><span class="spinner"></span></div>';
  try {
    const data = await api('/api/admin/health-check');
    el.innerHTML = renderHealth(data);
  } catch(e) {
    el.innerHTML = `<div class="empty" style="color:var(--red)">Error: ${e.message}</div>`;
  }
}

function renderHealth(data) {
  const overallOk = data.ready;
  const ts = data.timestamp ? new Date(data.timestamp).toLocaleString() : '—';

  const bigStatus = `
  <div class="big-status">
    <div class="big-dot" style="background:${overallOk ? 'var(--green)' : 'var(--red)'}; box-shadow:0 0 10px ${overallOk ? 'rgba(34,197,94,.5)' : 'rgba(239,68,68,.5)'}"></div>
    <div>
      <div style="font-size:16px;font-weight:700;color:${overallOk ? 'var(--green)' : 'var(--red)'}">
        ${overallOk ? '✅ All Systems Operational' : '❌ Issues Detected'}
      </div>
      <div style="font-size:12px;color:var(--muted);margin-top:2px">${esc(data.next_steps || '')} · Checked ${ts}</div>
    </div>
  </div>`;

  const systems = data.systems || {};

  // Special rendering for stripe_startup
  function cardClass(key, s) {
    const st = (s.status || '').toLowerCase();
    if (st.startsWith('ok')) return 'health-ok';
    if (st.startsWith('fail')) return 'health-fail';
    return 'health-warn';
  }

  function extraDetail(key, s) {
    if (key === 'stripe_startup') {
      return `<div class="health-card-detail">Validated: ${s.validated_at || '—'} · Escrow gate: <strong>${s.escrow_gate || '—'}</strong></div>`;
    }
    if (key === 'stripe_webhook') {
      return `<div class="health-card-detail">Endpoint: <span class="mono">${s.endpoint || '—'}</span> · Secret set: ${s.webhook_secret_set ? 'yes' : 'no'}</div>`;
    }
    if (key === 'marketplace') {
      return `<div class="health-card-detail">Live listings: ${s.live_listings ?? '—'}</div>`;
    }
    if (s.error) return `<div class="health-card-detail" style="color:var(--red)">${esc(s.error)}</div>`;
    return '';
  }

  const cards = Object.entries(systems).map(([key, s]) => `
    <div class="health-card">
      <div class="health-card-name">${key.replace(/_/g,' ')}</div>
      <div class="health-card-status ${cardClass(key, s)}">${esc(s.status || '—')}</div>
      ${extraDetail(key, s)}
    </div>`).join('');

  return bigStatus + `<div class="health-grid">${cards}</div>`;
}

// ── Utilities ─────────────────────────────────────────────────────────
function esc(str) {
  return String(str ?? '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
