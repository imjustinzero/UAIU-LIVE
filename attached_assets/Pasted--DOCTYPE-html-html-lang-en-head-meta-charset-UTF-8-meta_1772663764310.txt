// SHA-256 via Web Crypto (real cryptographic hash — not fake)
async function sha256(message) {
  const msgBuffer = new TextEncoder().encode(message);
  const hashBuffer = await crypto.subtle.digest('SHA-256', msgBuffer);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
}

  return 'UAIU-' + side + '-' + Date.now().toString().slice(-6) + Math.random().toString(36).slice(2,5).toUpperCase();
}

async function submitRFQ() {
  if (!rateOk('rfq', 3)) return;
  if (botDetected()) { showToast('⚠ Submission blocked.'); return; }

  const company = sanitize(document.getElementById('rfqCompany').value);
  const contact = sanitize(document.getElementById('rfqContact').value);
  const email = sanitize(document.getElementById('rfqEmail').value);
  const side = document.getElementById('rfqSide').value;
  const standard = document.getElementById('rfqStandard').value;
  const volume = parseInt(document.getElementById('rfqVolume').value);
  const price = parseFloat(document.getElementById('rfqPrice').value) || null;
  const origin = sanitize(document.getElementById('rfqOrigin').value);
  const vintage = parseInt(document.getElementById('rfqVintage').value) || null;
  const deadline = sanitize(document.getElementById('rfqDeadline').value);
  const notes = sanitize(document.getElementById('rfqNotes').value);

  if (!company || !contact || !email || !volume || volume < 1) {
    showToast('⚠ Please fill in all required fields.');
    return;
  }
  if (volume < 1000) {
    showToast('⚠ Minimum RFQ volume is 1,000 tonnes.');
    return;
  }
  if (!email.includes('@') || !email.includes('.')) {
    showToast('⚠ Please enter a valid email address.');
    return;
  }

  const btn = document.getElementById('rfqSubmitBtn');
  btn.textContent = '⏳ Submitting...'; btn.disabled = true;

  // Generate cryptographic receipt
  const rfqId = genTradeId(side);
  const payload = JSON.stringify({ rfqId, company, contact, email, side, standard, volume, price, origin, vintage, notes, ts: Date.now() });
  const hash = await sha256(payload + (_lastReceiptHash || 'GENESIS'));
  _lastReceiptHash = hash;

  const rfq = { id: rfqId, company, contact, email, side, standard, volume, price, origin, vintage, deadline, notes, hash, status: 'NEW', ts: Date.now() };
  _rfqs.unshift(rfq);
  _track('rfq_submit', rfqId);

  setTimeout(() => {
    document.getElementById('rfqFormBody').style.display = 'none';
    document.getElementById('rfqSuccess').style.display = 'block';
    document.getElementById('rfqRef').textContent = rfqId;
    btn.textContent = 'Submit RFQ — Receive Priced Quote →'; btn.disabled = false;

async function createTradeReceipt(tradeData) {
  const prevHash = _lastReceiptHash || 'GENESIS_BLOCK_UAIU_CARIBBEAN_CARBON_EXCHANGE';
  const payload = JSON.stringify({ ...tradeData, prev: prevHash, ts: Date.now() });
  const hash = await sha256(payload);
  _lastReceiptHash = hash;

  const trade = {
    ...tradeData,
    receiptHash: hash,
    prevReceiptHash: prevHash,
    ts: Date.now(),
    verifyUrl: 'uaiu.live/verify/' + tradeData.id,
    auditUrl: 'uaiu.live/audit/' + tradeData.id + '.pdf'
  };
  _trades.unshift(trade);
  return trade;
}

// 1. UNIQUE SESSION TOKEN (crypto-grade random)
const _sid=(()=>{
  try{
    const a=new Uint8Array(16);
    crypto.getRandomValues(a);
    return Array.from(a,b=>b.toString(16).padStart(2,'0')).join('');
  }catch(e){return Math.random().toString(36).slice(2)+Date.now().toString(36)}
})();
const _st=Date.now();
const _log=[];
function _track(t,d=''){
  _log.push({t,d:String(d).slice(0,120),ms:Date.now()-_st,sid:_sid});
  if(_log.length>200)_log.shift();
}

// 2. INPUT SANITIZER — strips XSS/injection from all inputs
function sanitize(v){
  return String(v||'')
    .replace(/<script[\s\S]*?>[\s\S]*?<\/script>/gi,'')
    .replace(/<[^>]+>/g,'')
    .replace(/javascript:/gi,'')
    .replace(/on\w+\s*=/gi,'')
    .replace(/[<>"'`\\]/g,c=>({'<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#x27;','`':'&#x60;','\\':'&#x5C;'}[c]))
    .trim().slice(0,500);
}
// Auto-sanitize on blur for every input/textarea
document.querySelectorAll('input[type=text],input[type=email],input[type=number],textarea').forEach(el=>{
  el.addEventListener('blur',()=>{if(el.value&&el.type!=='number')el.value=sanitize(el.value)});
});

// 3. RATE LIMITER — blocks form spam
const _rl={};
function rateOk(key,max=4){
  const now=Date.now();
  _rl[key]=(_rl[key]||[]).filter(t=>now-t<60000);
  if(_rl[key].length>=max){showToast('⚠ Too many requests. Please wait.');_track('rate_limit',key);return false;}
  _rl[key].push(now);return true;
}
